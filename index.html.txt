<script>
const game = {
    level: 1, best: 1, clicks: 1, angle: 0, dir: 1,
    target: { a: 0, w: 0.6 }, particles: [], state: 'MENU', pulse: 0,
    themes: [
        { bg: '#ffffff', acc: '#FFD700' }, { bg: '#f0faff', acc: '#007AFF' },
        { bg: '#fff0f0', acc: '#FF3B30' }, { bg: '#f0fff0', acc: '#34C759' },
        { bg: '#f5f0ff', acc: '#5856D6' }, { bg: '#111111', acc: '#ffffff' }
    ],

    init() {
        this.canvas = document.getElementById('gameCanvas');
        this.ctx = this.canvas.getContext('2d');
        this.resize();
        window.onresize = () => this.resize();
        
        // Kayıtları Yükle
        this.level = parseInt(localStorage.getItem('kilit_lvl')) || 1;
        this.best = parseInt(localStorage.getItem('kilit_best')) || 1;
        this.updateUI();
        this.loop();
    },

    resize() {
        this.width = this.canvas.width = window.innerWidth;
        this.height = this.canvas.height = window.innerHeight;
        this.radius = Math.min(this.width, this.height) * 0.33;
    },

    // --- SES MOTORU (Daha Tok Sesler) ---
    beep(type) {
        if(!this.audioCtx) this.audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        const osc = this.audioCtx.createOscillator();
        const gain = this.audioCtx.createGain();
        osc.connect(gain); gain.connect(this.audioCtx.destination);

        if(type === 'hit') {
            osc.frequency.setValueAtTime(440, this.audioCtx.currentTime); // Tok tık
            gain.gain.setValueAtTime(0.1, this.audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, this.audioCtx.currentTime + 0.1);
            osc.start(); osc.stop(this.audioCtx.currentTime + 0.1);
        } else if(type === 'fail') {
            osc.frequency.setValueAtTime(120, this.audioCtx.currentTime); // Hata
            osc.frequency.exponentialRampToValueAtTime(50, this.audioCtx.currentTime + 0.3);
            gain.gain.setValueAtTime(0.2, this.audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, this.audioCtx.currentTime + 0.3);
            osc.start(); osc.stop(this.audioCtx.currentTime + 0.3);
        }
    },

    start() {
        this.clicks = this.level;
        this.state = 'PLAY';
        document.getElementById('menu').style.opacity = '0';
        setTimeout(() => document.getElementById('menu').style.display = 'none', 500);
        this.setTarget();
        this.updateUI();
    },

    setTarget() {
        const gap = 1.6 + Math.random() * 2.4;
        this.target.a = (this.angle + (this.dir * gap)) % (Math.PI * 2);
        this.target.w = Math.max(0.25, 0.6 - (this.level * 0.004));
    },

    updateUI() {
        document.getElementById('ui-lvl').innerText = (this.level < 10 ? "0"+this.level : this.level);
        document.getElementById('ui-best').innerText = (this.best < 10 ? "0"+this.best : this.best);
        document.getElementById('remaining').innerText = this.clicks;
        
        const theme = this.themes[(this.level - 1) % this.themes.length];
        document.body.style.backgroundColor = theme.bg;
        // Karanlık modda metinleri beyaz yap
        const isDark = theme.bg === '#111111';
        document.querySelectorAll('.val, .label, h1, #remaining').forEach(el => el.style.color = isDark ? '#fff' : '#000');
    },

    vibrate(ms) { if (navigator.vibrate) navigator.vibrate(ms); },

    check() {
        if(this.state !== 'PLAY') return;

        const nA = (this.angle % 6.28 + 6.28) % 6.28;
        let s = (this.target.a % 6.28 + 6.28) % 6.28;
        let e = (s + this.target.w) % 6.28;
        let hit = (s < e) ? (nA >= s && nA <= e) : (nA >= s || nA <= e);

        if(hit) {
            this.beep('hit');
            this.vibrate(15);
            this.spawnParticles();
            this.pulse = 15; // Halka nabız efekti başlasın
            this.clicks--;
            
            if(this.clicks <= 0) {
                this.level++;
                this.clicks = this.level;
                if(this.level > this.best) {
                    this.best = this.level;
                    localStorage.setItem('kilit_best', this.best);
                }
                localStorage.setItem('kilit_lvl', this.level);
                this.vibrate([30, 30, 30]);
            }
            this.dir *= -1;
            this.setTarget();
            this.updateUI();
            
            // Rakam büyüme animasyonu
            const rem = document.getElementById('remaining');
            rem.style.transform = 'scale(1.2)';
            setTimeout(() => rem.style.transform = 'scale(1)', 150);

        } else {
            this.beep('fail');
            this.vibrate(100);
            this.state = 'MENU';
            document.getElementById('menu').style.display = 'flex';
            setTimeout(() => document.getElementById('menu').style.opacity = '1', 10);
            document.getElementById('menu-title').innerText = "FAILED.";
            document.getElementById('menu-title').style.color = "#FF3B30";
        }
    },

    spawnParticles() {
        const px = (this.width/2) + Math.cos(this.angle) * this.radius;
        const py = (this.height/2) + Math.sin(this.angle) * this.radius;
        const color = this.themes[(this.level - 1) % this.themes.length].acc;
        for(let i=0; i<12; i++) {
            this.particles.push({ x: px, y: py, vx: (Math.random()-0.5) * 8, vy: (Math.random()-0.5) * 8, l: 1.0, c: color });
        }
    },

    loop() {
        const cx = this.width / 2;
        const cy = this.height / 2;
        const theme = this.themes[(this.level - 1) % this.themes.length];
        this.ctx.clearRect(0, 0, this.width, this.height);

        // Nabız efektini hesapla
        let currentRadius = this.radius + (this.pulse * 0.5);
        if(this.pulse > 0) this.pulse -= 1;

        // 1. Arka Halka
        this.ctx.beginPath();
        this.ctx.arc(cx, cy, currentRadius, 0, 6.28);
        this.ctx.strokeStyle = theme.bg === '#111111' ? 'rgba(255,255,255,0.05)' : 'rgba(0,0,0,0.03)';
        this.ctx.lineWidth = 2;
        this.ctx.stroke();

        // 2. Hedef
        this.ctx.beginPath();
        this.ctx.arc(cx, cy, currentRadius, this.target.a, this.target.a + this.target.w);
        this.ctx.strokeStyle = theme.acc;
        this.ctx.lineWidth = 6;
        this.ctx.lineCap = 'round';
        this.ctx.stroke();

        // 3. Tozlar
        this.particles.forEach((p, i) => {
            p.x += p.vx; p.y += p.vy; p.l -= 0.03;
            this.ctx.fillStyle = p.c;
            this.ctx.globalAlpha = p.l;
            this.ctx.fillRect(p.x, p.y, 2, 2);
            if(p.l <= 0) this.particles.splice(i, 1);
        });
        this.ctx.globalAlpha = 1;

        // 4. İmleç (Nokta)
        if(this.state === 'PLAY') {
            this.angle += (0.035 + (this.level * 0.0002)) * this.dir;
        }
        const dotX = cx + Math.cos(this.angle) * currentRadius;
        const dotY = cy + Math.sin(this.angle) * currentRadius;
        this.ctx.beginPath();
        this.ctx.arc(dotX, dotY, 4, 0, 6.28);
        this.ctx.fillStyle = theme.bg === '#111111' ? '#fff' : '#000';
        this.ctx.fill();

        requestAnimationFrame(() => this.loop());
    }
};

window.onload = () => game.init();
const handleInput = (e) => { if(e.type === 'touchstart') e.preventDefault(); if(game.state === 'PLAY') game.check(); };
window.addEventListener('touchstart', handleInput, {passive:false});
window.addEventListener('mousedown', handleInput);
</script>
